import ssl

df = pd.read_csv("https://raw.githubusercontent.com/drobaina/Python/main/microdados_ed_basica_RJ_2022.csv", header=0)
df.head()

df_rj = df[df["NO_MUNICIPIO"] == "Rio de Janeiro"]

# Selecionar apenas as colunas "TP_DEPENDENCIA" e "TP_LOCALIZACAO"
## TP_DEPENDENCIA -> Dependncia Administrativa
## TP_LOCALIZACAO -> Rural ou Urbana
nova_base = df_rj[["TP_DEPENDENCIA", "TP_LOCALIZACAO"]]

print(nova_base.head())

tipo_adm = nova_base.groupby('TP_DEPENDENCIA').size().reset_index(name='Tipo de Administraao')
tipo_adm

area = nova_base.groupby('TP_LOCALIZACAO').size().reset_index(name='AREA')
area
